2. Le vrai problème : créer un résolveur central
Le fond du problème reste que rien ne consulte ces tables. Il faut un seul fichier qui fait office de "vérité" pour toute URL, et que tous les composants de page interrogent — au lieu d'afficher location.pathname brut.
ts// app/i18n/resolve.ts
import { STATIC_ROUTES, type Lang } from "./routes";
import { DYNAMIC_GROUPS } from "../data/dynamic-routes";

export interface ResolvedPage {
  type: "static" | "dynamic";
  key: string;        // ex: "contact" ou "celebrity_hair_transplant_en"
  slug?: string;       // uniquement pour dynamic, ex: "elon-musk-hair-transplant"
  lang: Lang;
}

// path = ce qui suit le préfixe de langue, TOUJOURS avec un "/" en tête, sans slash final (sauf "/")
export function resolvePage(lang: Lang, path: string): ResolvedPage | null {
  const normalized = path === "" ? "/" : path.replace(/\/$/, "") || "/";

  for (const route of STATIC_ROUTES) {
    if (route.langs[lang] === normalized) {
      return { type: "static", key: route.key, lang };
    }
  }

  for (const group of DYNAMIC_GROUPS) {
    for (const variant of group.variants) {
      if (variant.langs[lang] === normalized) {
        return { type: "dynamic", key: group.key, slug: variant.slug, lang };
      }
    }
  }

  return null; // vraie 404 — aucune correspondance dans les tables
}

tsx// app/routes/en-pages.tsx
import { useLocation } from "react-router";
import { resolvePage } from "../i18n/resolve";
import { NotFoundPage } from "./not-found-page";

export function meta() {
  return [{ title: "Page" }];
}

export default function EnPages() {
  const location = useLocation();
  const resolved = resolvePage("en", location.pathname);

  if (!resolved) return <NotFoundPage path={location.pathname} />;

  return (
    <div style={page}>
      <div style={card}>
        <span style={badge}>EN</span>
        <h1 style={title}>{resolved.key}</h1>
        <p style={sub}>
          Type: <code style={code}>{resolved.type}</code>
          {resolved.slug && <> · Slug: <code style={code}>{resolved.slug}</code></>}
        </p>
        <p style={hint}>Placeholder — content will be added in the next phase.</p>
      </div>
    </div>
  );
}
